# Post-Order Feedback Request — Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** After a master completes an order, automatically send the client a 1–5 rating request via client_bot after a configurable delay; route the response to a thank-you with review links (5), a soft follow-up + master notification (4), or an alert to master (≤3).

**Architecture:** Polling scheduler job (same pattern as 24h/1h reminders) runs every 30 minutes and queries orders where `status='done'`, `feedback_sent=FALSE`, and `done_at + feedback_delay_hours * 3600 ≤ now()`. Client responds via inline keyboard; callback handler routes by score and notifies master if needed.

**Tech Stack:** Python/asyncio, aiogram 3.x, SQLite (aiosqlite), APScheduler, FastAPI, React (Mini App)

**Design doc:** `docs/plans/2026-04-17-feedback-design.md`

```python
async def get_orders_for_feedback() -> list[dict]:
    """Get completed orders that need a feedback request.

    Criteria:
    - status = 'done'
    - feedback_sent = FALSE
    - done_at + feedback_delay_hours * 3600 seconds <= now()
    - client has tg_id
    """
    conn = await get_connection()
    try:
        cursor = await conn.execute(
            """
            SELECT
                o.id            AS order_id,
                o.done_at,
                c.tg_id         AS client_tg_id,
                c.name          AS client_name,
                m.id            AS master_id,
                m.tg_id         AS master_tg_id,
                m.name          AS master_name,
                COALESCE(m.feedback_delay_hours, 3)  AS feedback_delay_hours,
                m.feedback_message,
                m.feedback_reply_5,
                m.review_buttons,
                GROUP_CONCAT(oi.name, ', ')           AS services
            FROM orders o
            JOIN clients c  ON o.client_id  = c.id
            JOIN masters m  ON o.master_id  = m.id
            LEFT JOIN order_items oi ON o.id = oi.order_id
            WHERE o.status        = 'done'
              AND o.feedback_sent = 0
              AND c.tg_id         IS NOT NULL
              AND o.done_at       IS NOT NULL
              AND (julianday('now') - julianday(o.done_at)) * 24
                  >= COALESCE(m.feedback_delay_hours, 3)
            GROUP BY o.id
            """
        )
        rows = await cursor.fetchall()
        return [dict(row) for row in rows]
    finally:
        await conn.close()


async def mark_feedback_sent(order_id: int) -> None:
    """Mark that a feedback request was sent for this order."""
    conn = await get_connection()
    try:
        await conn.execute(
            "UPDATE orders SET feedback_sent = 1 WHERE id = ?",
            (order_id,)
        )
        await conn.commit()
    finally:
        await conn.close()


async def save_order_rating(order_id: int, rating: int) -> None:
    """Save client rating (1-5) for a completed order."""
    conn = await get_connection()
    try:
        await conn.execute(
            "UPDATE orders SET rating = ? WHERE id = ?",
            (rating, order_id)
        )
        await conn.commit()
    finally:
        await conn.close()
```

```python
def render_feedback_message(template: Optional[str], default: str, master_name: str, services: str) -> str:
    """Render feedback message template with variable substitution."""
    text = template if template else default
    replacements = {
        "{master_name}": master_name,
        "{мастер}": master_name,
        "{service}": services,
        "{услуга}": services,
    }
    for key, value in replacements.items():
        text = text.replace(key, value)
    return text
```

```python
async def send_feedback_requests(client_bot: Bot) -> None:
    """Send post-order feedback requests to clients."""
    logger.info("Running feedback request task")
    try:
        orders = await get_orders_for_feedback()
        logger.info(f"Found {len(orders)} orders for feedback request")

        for order in orders:
            try:
                text = render_feedback_message(
                    template=order.get("feedback_message"),
                    default=DEFAULT_FEEDBACK_MESSAGE,
                    master_name=order.get("master_name") or "мастера",
                    services=order.get("services") or "—",
                )
                await client_bot.send_message(
                    chat_id=order["client_tg_id"],
                    text=text,
                    reply_markup=feedback_rating_kb(order["order_id"]),
                )
                await mark_feedback_sent(order["order_id"])
                logger.info(f"Sent feedback request for order {order['order_id']}")

            except TelegramForbiddenError:
                logger.warning(f"Client {order['client_tg_id']} blocked the bot, skipping")
                await mark_feedback_sent(order["order_id"])
            except TelegramBadRequest as e:
                logger.error(f"Failed to send feedback request for order {order['order_id']}: {e}")
            except Exception as e:
                logger.error(f"Error sending feedback request for order {order['order_id']}: {e}")

    except Exception as e:
        logger.error(f"Error in send_feedback_requests: {e}")
```

```python
@router.callback_query(lambda c: c.data and c.data.startswith("feedback:"))
async def handle_feedback_rating(callback: CallbackQuery) -> None:
    """Handle client rating response from post-order feedback."""
    try:
        _, order_id_str, rating_str = callback.data.split(":")
        order_id = int(order_id_str)
        rating = int(rating_str)
    except (ValueError, AttributeError):
        await callback.answer()
        return

    await save_order_rating(order_id, rating)

    # Remove inline keyboard by editing the original message
    try:
        await callback.message.edit_reply_markup(reply_markup=None)
    except Exception:
        pass

    if rating == 5:
        # Fetch master's review_buttons
        order = await get_order_by_id_for_feedback(order_id)
        reply_text = render_feedback_message(
            template=order.get("feedback_reply_5") if order else None,
            default=DEFAULT_FEEDBACK_REPLY_5,
            master_name=order.get("master_name") or "" if order else "",
            services="",
        )
        review_buttons_raw = order.get("review_buttons") if order else None
        reply_markup = None
        if review_buttons_raw:
            try:
                import json
                buttons = json.loads(review_buttons_raw)
                rows = [
                    [InlineKeyboardButton(text=b["label"], url=b["url"])]
                    for b in buttons[:3]
                    if b.get("label") and b.get("url")
                ]
                if rows:
                    reply_markup = InlineKeyboardMarkup(inline_keyboard=rows)
            except Exception:
                pass
        await callback.message.answer(reply_text, reply_markup=reply_markup)

    elif rating == 4:
        await callback.message.answer("Расскажите, что можно улучшить? Это поможет стать лучше.")
        # Notify master
        order = await get_order_by_id_for_feedback(order_id)
        if order and order.get("master_tg_id"):
            client_name = order.get("client_name") or "Клиент"
            try:
                await master_bot_instance.send_message(
                    order["master_tg_id"],
                    f"ℹ️ {client_name} оценил визит на 4.\nЗаказ #{order_id}",
                )
            except Exception as e:
                logger.warning(f"Failed to notify master for feedback rating 4, order {order_id}: {e}")

    else:  # rating <= 3
        await callback.message.answer("Мы свяжемся с вами в ближайшее время.")
        # Alert master
        order = await get_order_by_id_for_feedback(order_id)
        if order and order.get("master_tg_id"):
            client_name = order.get("client_name") or "Клиент"
            try:
                await master_bot_instance.send_message(
                    order["master_tg_id"],
                    f"⚠️ {client_name} недоволен!\nОценка: {rating} из 5\nЗаказ #{order_id}\n\nСвяжитесь с клиентом.",
                )
            except Exception as e:
                logger.warning(f"Failed to alert master for feedback rating {rating}, order {order_id}: {e}")

    await callback.answer()
```

```python
class FeedbackSettingsBody(BaseModel):
    feedback_delay_hours: int = 3
    feedback_message: Optional[str] = None
    feedback_reply_5: Optional[str] = None
    review_buttons: Optional[list[dict]] = None  # [{label: str, url: str}]

    @field_validator("feedback_delay_hours")
    @classmethod
    def delay_in_range(cls, v):
        if not (1 <= v <= 72):
            raise ValueError("feedback_delay_hours must be between 1 and 72")
        return v

    @field_validator("review_buttons")
    @classmethod
    def max_three_buttons(cls, v):
        if v is not None:
            if len(v) > 3:
                raise ValueError("review_buttons: max 3 allowed")
            for btn in v:
                url = btn.get("url", "")
                if not (url.startswith("http://") or url.startswith("https://")):
                    raise ValueError(f"review_buttons: invalid URL '{url}'")
        return v
```

```jsx
const WebApp = window.Telegram?.WebApp;

function haptic(type = 'light') {
  WebApp?.HapticFeedback?.impactOccurred?.(type);
}

export default function FeedbackSettings({ onBack }) {
  const qc = useQueryClient();
  const [successMsg, setSuccessMsg] = useState('');

  const { data, isLoading } = useQuery({
    queryKey: ['master-feedback-settings'],
    queryFn: getMasterFeedbackSettings,
    staleTime: 60_000,
  });

  const [message, setMessage] = useState('');
  const [reply5, setReply5] = useState('');

    setDelayHours(String(data.feedback_delay_hours ?? 3));
    setMessage(data.feedback_message ?? '');
    setReply5(data.feedback_reply_5 ?? '');

            className="onb-input"
            type="number"
            min={1}
            max={72}
            value={delayHours}
            onChange={e => setDelayHours(e.target.value)}
          />
        </div>

      <div className="enterprise-cell-group">
        <div className="onb-field-group" style={{ padding: '0 16px 12px' }}>
          <label className="onb-label">Текст запроса (переменные: {'{master_name}'}, {'{service}'})</label>
```
